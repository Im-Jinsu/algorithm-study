import java.util.*;
class Solution {
    public int[] solution(int[] progresses, int[] speeds) {
        double DEPLOY_PERCENT=100.0;
        double day = 0.0;
        int[] days = new int[speeds.length];
        
        for(int i=0 ; i<speeds.length ; i++){
            day = (DEPLOY_PERCENT - progresses[i])/speeds[i];
            if(day%1!=0) day= day+1;
            days[i]= (int)day;
        }
        
        int maxDay = days[0];
        int count=1;
        List<Integer> deployeDays = new ArrayList<>();
        for(int i=1 ; i<days.length ; i++){
            if(maxDay>=days[i])
                count++;
            if(maxDay<days[i]){
                deployeDays.add(count);
                count=1;
            }
        }
        
        deployeDays.add(count);
        int[] answer = new int[deployeDays.size()];
        for(int i=0 ; i<deployeDays.size() ; i++)
            answer[i] = deployeDays.get(i);
        
        return answer;
    }
}
