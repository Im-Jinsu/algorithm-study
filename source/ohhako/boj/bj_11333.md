### 접근 방법을 적용한 코드

```java
import java.util.Scanner;

public class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int t = sc.nextInt();

        for(int i=0 ; i<t ; i++){
            int n = sc.nextInt();
            int[] dp = new int[n+1];
            dp[0] =3;
            if(n%3==0){
                for(int j=3 ; j<=n ; j+=3){
                    dp[j] = (dp[j-3]*3)%1000000007;
                    if(j==6){
                        dp[j] += (j%3*2);
                    }else{
                        for(int k=j-6 ; k>0 ; k=-3){
                            dp[j] = (dp[j]+(dp[k]*(k+3)%3*2))%1000000007;
                        }
                    }
                }
            }
            System.out.println(dp[n]);
        }
    }
}
```
